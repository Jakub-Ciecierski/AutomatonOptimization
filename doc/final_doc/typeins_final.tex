%%
%	AddFourPlots
%
% #1 = C4 Training Small
% #2 = C4 Training Big
% #3 = C4 Training All
% #4 = C4 Testing
%
% #5 = C5 Training Small
% #6 = C5 Training Big
% #7 = C5 Training All
% #8 = C5 Testing
%
%%
\newcommand{\addFourPlots}[8]{
    		% Training |w| < C    
        \addplot[style={bblue,fill=bblue,mark=none}]
            coordinates 
            {
            		(C=4, #1)
        			(C=5, #5)
       		};

    		% Training |w| > C
       	\addplot[style={rred,fill=rred,mark=none}]
            coordinates 
            {
            		(C=4, #2)
        			(C=5, #6)
       		};
       		       		
		% Training All	
        \addplot[style={ggreen,fill=ggreen,mark=none}]
            coordinates 
            {
            		(C=4, #3)
        			(C=5, #7)
       		};

		% Testing
       	\addplot[style={ppurple,fill=ppurple,mark=none}]
            coordinates 
            {
            		(C=4, #4)
        			(C=5, #8)
       		};
}
